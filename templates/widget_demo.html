Quick Start

Add this code snippet to your restaurant website to enable AI-powered table reservations:

<!-- Pareto Booking Widget -->
<script src="{{ request.url_root }}widget/embed.js"></script>
<script>
  ParetoBooking.init({
    assistantId: 'rest_demo_123',      // Your unique restaurant ID
    position: 'bottom-right',          // Widget position
    primaryColor: '#4CAF50',          // Your brand color
    headerTitle: 'Table Reservation',  // Widget header
    welcomeMessage: 'Hi! Book a table at our restaurant.'
  });
</script>
